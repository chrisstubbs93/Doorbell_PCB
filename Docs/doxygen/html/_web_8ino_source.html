103 boolean connectWiFi() {
104  int connAttempts = 0;
105  while (connAttempts < 3) {
106  connAttempts++;
107  if (EAP_IDENTITY == "") {
108  //if wpa2-psk is configured
109  Serial.println((String)"[INFO] WPA2-PSK mode.");
110  WiFi.begin(ssid.c_str(), rot(wifipsk).c_str());
111  } else {
112  //if eduroam is configured
113  Serial.println((String)"[INFO] WPA2-Enterprise mode.");
114  WiFi.disconnect(true);
115  WiFi.mode(WIFI_STA);
116  const unsigned char* EAP_IDENTITY_A = reinterpret_cast<const unsigned char *>( rot(EAP_IDENTITY).c_str() ); //convert strings to unsigned char for enterprise functions
117  const unsigned char* EAP_PASSWORD_A = reinterpret_cast<const unsigned char *>( rot(EAP_PASSWORD).c_str() );
118  esp_wifi_sta_wpa2_ent_set_username(EAP_IDENTITY_A, strlen(rot(EAP_IDENTITY).c_str())); //set enterprise wifi parameters
119  esp_wifi_sta_wpa2_ent_set_password(EAP_PASSWORD_A, strlen(rot(EAP_PASSWORD).c_str()));
120  esp_wpa2_config_t config = WPA2_CONFIG_INIT_DEFAULT(); //set config settings to default
121  esp_wifi_sta_wpa2_ent_enable(&config); //enable enterprise mode
122  WiFi.begin(ssid.c_str(), rot(wifipsk).c_str()); //connect to AP
123  }
124  Serial.print("[INFO] Connecting to " + ssid + ". Attempt " + connAttempts);
125  addToLog(ssid + " connecting");
126  if (checkConnection()) {
127  return true; //Connected OK.
128  } else {
129  //Connection failed. Try again.
130  if (connAttempts == 3) {
131  //Display last attempt error
132  errorMsg("WiFi connection timed out.\nThis is attempt " + (String)connAttempts + "/3.\nAborting connection.\nEntering setup mode.");
133  } else {
134  //Display retry error
135  errorMsg("WiFi connection timed out.\nThis is attempt " + (String)connAttempts + "/3.\nTrying again in 10s.\nPress [A] to retry now.\nPress [B] to run setup.");
136  unsigned long tmr = millis();
137  while ((millis() - tmr < 10000)) {
138  if (digitalRead(BTNAPIN)) {
139  //User requested retry
140  if (connAttempts == 3) connAttempts = 2; //Manually allow one more try if on last try.
141  break;
142  }
143  if (digitalRead(BTNBPIN)) {
144  //User requested setup mode
145  settingMode = true;
146  return false;
147  }
148  }
149  }
150  settingMode = true;
151  }
152  }
153  //Failed to connect after 3 attempts.
154  return false;
155 }
